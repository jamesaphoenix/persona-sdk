# 📊 Market Research That Saves You $100K

Stop paying for expensive surveys. Generate responses that match real customer behavior in minutes.

## Your First Survey in 2 Minutes ⚡

```typescript
import { PersonaGroup, StructuredOutputGenerator } from '@jamesaphoenix/persona-sdk';
import { NormalDistribution, BetaDistribution } from '@jamesaphoenix/persona-sdk/distributions';
import { z } from 'zod';

// Define your target market
const targetMarket = await PersonaGroup.generate({
  size: 1000,
  segments: [
    {
      name: "Enterprise Buyers",
      weight: 0.2,
      attributes: {
        company_size: new NormalDistribution(500, 200),
        budget: new NormalDistribution(50000, 15000),
        decision_power: new BetaDistribution(4, 2), // High influence
        pain_points: ['scalability', 'integration', 'cost']
      }
    },
    {
      name: "SMB Owners",
      weight: 0.5,
      attributes: {
        company_size: new NormalDistribution(50, 30),
        budget: new NormalDistribution(5000, 2000),
        decision_power: new BetaDistribution(5, 1), // Direct decision makers
        pain_points: ['cost', 'ease-of-use', 'support']
      }
    },
    {
      name: "Startup Founders",
      weight: 0.3,
      attributes: {
        company_size: new NormalDistribution(10, 5),
        budget: new NormalDistribution(1000, 500),
        decision_power: new BetaDistribution(5, 1),
        pain_points: ['speed', 'flexibility', 'scaling']
      }
    }
  ]
});

// Design your survey
const surveyQuestions = {
  product: "AI-Powered Analytics Platform",
  pricing_tiers: ["$99/mo", "$299/mo", "$999/mo", "Enterprise"],
  features: [
    "Real-time dashboards",
    "Predictive analytics", 
    "Custom integrations",
    "White-label options",
    "24/7 support"
  ]
};

const generator = new StructuredOutputGenerator();
```

## Generate Realistic Survey Responses 🎯

```typescript
const SurveyResponseSchema = z.object({
  segment_breakdown: z.array(z.object({
    segment: z.string(),
    response_count: z.number(),
    response_rate: z.number()
  })),
  
  product_interest: z.object({
    very_interested: z.number(),
    interested: z.number(),
    neutral: z.number(),
    not_interested: z.number(),
    strongly_not_interested: z.number()
  }),
  
  pricing_sensitivity: z.object({
    preferred_tier: z.record(z.string(), z.number()),
    max_willingness_to_pay: z.object({
      mean: z.number(),
      p25: z.number(),
      p50: z.number(),
      p75: z.number()
    }),
    price_elasticity: z.number()
  }),
  
  feature_priorities: z.array(z.object({
    feature: z.string(),
    importance_score: z.number(),
    willing_to_pay_extra: z.boolean(),
    segment_variations: z.record(z.string(), z.number())
  })),
  
  purchase_likelihood: z.object({
    within_30_days: z.number(),
    within_90_days: z.number(),
    within_year: z.number(),
    never: z.number()
  }),
  
  objections: z.array(z.object({
    objection: z.string(),
    frequency: z.number(),
    severity: z.enum(['minor', 'moderate', 'dealbreaker']),
    affected_segments: z.array(z.string())
  })),
  
  competitive_analysis: z.object({
    current_solutions: z.array(z.object({
      competitor: z.string(),
      market_share: z.number(),
      switching_likelihood: z.number()
    })),
    switching_factors: z.array(z.string())
  })
});

const surveyResults = await generator.generateCustom(
  targetMarket,
  SurveyResponseSchema,
  `Generate market research survey responses for: ${JSON.stringify(surveyQuestions)}`
);
```

## Dynamic Persona-Based Focus Groups 🎤

```typescript
const FocusGroupSchema = z.object({
  participants: z.array(z.object({
    persona_id: z.string(),
    segment: z.string(),
    key_quote: z.string(),
    sentiment: z.enum(['positive', 'neutral', 'negative']),
    influence_level: z.enum(['champion', 'supporter', 'neutral', 'detractor'])
  })),
  
  discussion_themes: z.array(z.object({
    theme: z.string(),
    mentions: z.number(),
    sentiment: z.number().min(-1).max(1),
    example_quotes: z.array(z.string())
  })),
  
  consensus_points: z.array(z.string()),
  controversy_points: z.array(z.string()),
  
  actionable_insights: z.array(z.object({
    insight: z.string(),
    confidence: z.enum(['high', 'medium', 'low']),
    supporting_evidence: z.array(z.string())
  }))
});

// Run a virtual focus group
const focusGroup = await generator.generateCustom(
  targetMarket.filter(p => p.attributes.product_interest > 0.7), // Interested personas only
  FocusGroupSchema,
  `Conduct focus group discussion about ${surveyQuestions.product}`
);

// Get quotes you can use in marketing
focusGroup.data.participants
  .filter(p => p.influence_level === 'champion')
  .forEach(champion => {
    console.log(`"${champion.key_quote}" - ${champion.segment}`);
  });
```

## Statistical Validation That Matters 📊

```typescript
// Ensure your data is statistically sound
const ValidationSchema = z.object({
  sample_size_adequacy: z.boolean(),
  confidence_level: z.number(),
  margin_of_error: z.number(),
  statistical_power: z.number(),
  segment_representation: z.record(z.string(), z.boolean()),
  bias_warnings: z.array(z.string())
});

const validation = await generator.generateCustom(
  targetMarket,
  ValidationSchema,
  "Validate statistical significance of survey results"
);

if (!validation.data.sample_size_adequacy) {
  console.warn("⚠️ Sample size too small for conclusions");
}
```
