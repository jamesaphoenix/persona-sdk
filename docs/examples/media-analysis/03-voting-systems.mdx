```typescript
const generator = new StructuredOutputGenerator();

const MediaVotingSchema = z.object({
  mediaId: z.string(),
  predictions: z.object({
    upvotes: z.number(),
    downvotes: z.number(),
    ratio: z.number().min(0).max(1),
    final_score: z.number(),
    will_hit_frontpage: z.boolean(),
    peak_position: z.number().nullable()
  }),
  timeline: z.object({
    hour_1: z.object({ upvotes: z.number(), momentum: z.string() }),
    hour_6: z.object({ upvotes: z.number(), momentum: z.string() }),
    hour_24: z.object({ upvotes: z.number(), momentum: z.string() })
  }),
  demographics: z.object({
    top_supporters: z.array(z.object({
      segment: z.string(),
      percentage: z.number(),
      reasoning: z.string()
    })),
    detractors: z.array(z.object({
      segment: z.string(),
      percentage: z.number(),
      reasoning: z.string()
    }))
  }),
  engagement_metrics: z.object({
    comment_count: z.number(),
    award_likelihood: z.number(),
    share_rate: z.number(),
    save_rate: z.number()
  }),
  optimization_tips: z.array(z.string())
});

// Analyze all media items
const votingResults = await Promise.all(
  mediaItems.map(item => 
    generator.generateCustom(
      votingAudience,
      MediaVotingSchema,
      `Predict Reddit-style voting for: ${JSON.stringify(item)}`
    )
  )
);
```

## Advanced: Cross-Platform Voting Prediction üåê

```typescript
const platforms = ['reddit', 'hackernews', 'producthunt', 'devto'];

const CrossPlatformSchema = z.object({
  platform: z.string(),
  success_probability: z.number(),
  expected_score: z.number(),
  best_time_to_post: z.string(),
  title_adjustments: z.string(),
  platform_specific_tips: z.array(z.string())
});

// Find the best platform for each piece of content
const platformAnalysis = await Promise.all(
  mediaItems.map(async (item) => {
    const analyses = await Promise.all(
      platforms.map(platform =>
        generator.generateCustom(
          votingAudience,
          CrossPlatformSchema,
          `Analyze ${item.title} for ${platform}`
        )
      )
    );
    
    return {
      content: item.title,
      bestPlatform: analyses.sort((a, b) => 
        b.data.success_probability - a.data.success_probability
      )[0]
    };
  })
);

console.log("üèÜ Optimal Platform Matches:");
platformAnalysis.forEach(({ content, bestPlatform }) => {
  console.log(`"${content}" ‚Üí ${bestPlatform.data.platform} (${(bestPlatform.data.success_probability * 100).toFixed(0)}% success)`);
});
```
