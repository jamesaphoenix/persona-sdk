```typescript
import { PersonaGroup, StructuredOutputGenerator } from '@jamesaphoenix/persona-sdk';
import { z } from 'zod';

// 1. Create your audience
const audience = new PersonaGroup('My Target Market');
await audience.generateFromDistributions(1000, {
  age: new NormalDistribution(32, 8),
  income: new UniformDistribution(40000, 150000),
  interests: ['SaaS', 'productivity', 'automation'],
  pain_points: ['time management', 'tool overload', 'cost']
});

// 2. Fire up the AI
const generator = new StructuredOutputGenerator();

// 3. Get instant insights
const MarketingInsightSchema = z.object({
  targetAudience: z.string(),
  keyTraits: z.array(z.string()),
  marketingAdvice: z.string(),
  tagline_suggestions: z.array(z.string()),
  channel_recommendations: z.array(z.object({
    channel: z.string(),
    reasoning: z.string(),
    expected_roi: z.enum(['low', 'medium', 'high', 'exceptional'])
  }))
});

const insights = await generator.generateCustom(
  audience,
  MarketingInsightSchema,
  "Create a complete marketing strategy"
);

console.log(insights.data.marketingAdvice);
// "Focus on time-saving benefits. Your audience is overwhelmed professionals 
// who value efficiency over features. Lead with ROI, not technology."
```

```typescript
// Analyze different audience segments separately
const segments = {
  enterprise: audience.filter(p => p.attributes.company_size > 1000),

      segment,

      `Create go-to-market strategy for ${name} segment`

// Focus on your highest-value segments
const bestSegment = segmentStrategies.sort((a, b) => 
  (b.expected_ltv / b.expected_cac) - (a.expected_ltv / a.expected_cac)
```

## Chain Multiple Analyses for Complete Strategy ðŸ”—

```typescript
// Build a complete go-to-market strategy
async function buildGoToMarketStrategy(audience: PersonaGroup) {
  const generator = new StructuredOutputGenerator();
  
  // 1. Market Analysis
  const market = await generator.generateDistributionInsights(
    audience, 
    ['income', 'age', 'tech_savviness']
  );
  
  // 2. Segmentation
  const segments = await generator.generateSegments(audience, 4);
  
  // 3. Positioning for each segment
  const positioning = await Promise.all(
    segments.data.segments.map(segment => 
      generator.generateCustom(
        audience,
        PositioningSchema,
        `Position for ${segment.name}`
      )
    )
  );
  
  // 4. Campaign strategy
  const campaigns = await generator.generateCustom(
    audience,
    CampaignIdeaSchema,
    "Design launch campaign"
  );
  
  // 5. Success metrics
  const MetricsSchema = z.object({
    north_star_metric: z.string(),
    leading_indicators: z.array(z.string()),
    lagging_indicators: z.array(z.string()),
    target_values: z.record(z.string(), z.number()),
    measurement_plan: z.string()
  });
  
  const metrics = await generator.generateCustom(
    audience,
    MetricsSchema,
    "Define success metrics"
  );
  
  return {
    market,
    segments,
    positioning,
    campaigns,
    metrics
  };
}

// Get your complete GTM strategy in under 30 seconds
const gtmStrategy = await buildGoToMarketStrategy(audience);
```
