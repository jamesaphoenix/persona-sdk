### 3. Statistical Distributions = Real Behavior

Stop using fake data. Model reality:

```typescript
import { 
  NormalDistribution,      // Bell curve (age, income)
  UniformDistribution,     // Equal probability (random selection)
  BetaDistribution,       // Probability ranges (conversion rates)
  ExponentialDistribution, // Time-based events (churn, purchases)
  CorrelatedDistribution  // Related attributes (age â†” income)
} from '@jamesaphoenix/persona-sdk/distributions';

// Realistic correlated attributes
const realistic = new CorrelatedDistribution({
  age: new NormalDistribution(30, 10),
  income: new NormalDistribution(50000, 20000),
  experience: new UniformDistribution(0, 20)
});

// Income increases with age (realistic!)
realistic.addCorrelation('income', 'age', 0.6);
realistic.addConditional({
  attribute: 'income',
  dependsOn: 'age',
  transform: (income, age) => income * (1 + (age - 25) * 0.02)
});
```

import { PersonaGroup, StructuredOutputGenerator } from '@jamesaphoenix/persona-sdk';
import { z } from 'zod';

async function analyzeCompetitor(competitorName: string) {

      key_message: z.string(),

  const analysis = await generator.generateCustom(
    theirUsers,
    CompetitorAnalysisSchema,
    `How to win customers from ${competitorName}`
