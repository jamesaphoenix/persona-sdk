Quick Start

// Generate personas
import { PersonaBuilder, NormalDistribution } from '@jamesaphoenix/persona-sdk';

const persona = PersonaBuilder.create()
  .withName('Alice Johnson')
  .withAge(28)
  .withOccupation('Software Engineer')
  .withSex('female')
  .build();
